stage('Create Jira Ticket & Attach Artifact') {
    when {
        expression { params.Create_Jira_Ticket && params.target?.trim() }
    }
    environment {
        JIRA_URL     = 'https://jira.company.com'
        PROJECT_KEY  = 'PROJ'
        ISSUE_TYPE   = 'Task'
        ATTACHMENT   = "report_${BUILD_NUMBER}.html"
    }
    steps {
        withCredentials([string(credentialsId: 'jira-bearer-token', variable: 'JIRA_TOKEN')]) {
            script {
                sh '''
                    set -euo pipefail

                    # Ensure BUILD_TIMESTAMP exists
                    BUILD_TIMESTAMP="${BUILD_TIMESTAMP:-$(date '+%Y-%m-%d %H:%M:%S')}"

                    # Multi-line Jira description
                    DESCRIPTION="This ticket was automatically created by the CI/CD pipeline to track the results of an Acunetix security scan.\\n\\nScan Details:\\n- Tool: Acunetix\\n- Build Number: #${BUILD_NUMBER}\\n- Job Name: ${JOB_NAME}\\n- Build URL: ${BUILD_URL}\\n- Target ID: ${target}\\n- Execution Time: ${BUILD_TIMESTAMP}\\n\\nThe attached report contains the full vulnerability findings and evidence generated during this scan.\\n\\nPlease review the findings and take appropriate remediation actions where required."

                    echo "Creating Jira ticket..."

                    # Create the Jira ticket
                    RESPONSE=$(jq -n \
                        --arg summary "Acunetix Security Scan Report – Build #${BUILD_NUMBER}" \
                        --arg desc "$DESCRIPTION" \
                        '{
                            fields: {
                                project: { key: "PROJ" },
                                summary: $summary,
                                description: $desc,
                                issuetype: { name: "Task" }
                            }
                        }' \
                        | curl -s -w "\\n%{http_code}" -X POST \
                            -H "Authorization: Bearer $JIRA_TOKEN" \
                            -H "Content-Type: application/json" \
                            "$JIRA_URL/rest/api/2/issue" \
                            -d @-)

                    # Split response and HTTP code
                    BODY=$(echo "$RESPONSE" | head -n -1)
                    STATUS=$(echo "$RESPONSE" | tail -n 1)

                    if [ "$STATUS" -ne 201 ]; then
                        echo "❌ Jira issue creation failed (HTTP $STATUS)"
                        echo "$BODY"
                        exit 1
                    fi

                    ISSUE_KEY=$(echo "$BODY" | jq -r '.key')

                    if [ -z "$ISSUE_KEY" ] || [ "$ISSUE_KEY" = "null" ]; then
                        echo "❌ Failed to extract Jira issue key"
                        echo "$BODY"
                        exit 1
                    fi

                    echo "✅ Created Jira issue: $ISSUE_KEY"

                    # Validate attachment
                    if [ ! -f "$ATTACHMENT" ]; then
                        echo "❌ Attachment file not found: $ATTACHMENT"
                        exit 1
                    fi

                    echo "Uploading attachment..."

                    ATTACH_STATUS=$(curl -s -o /dev/null -w "%{http_code}" -X POST \
                        -H "Authorization: Bearer $JIRA_TOKEN" \
                        -H "X-Atlassian-Token: no-check" \
                        -F "file=@$ATTACHMENT" \
                        "$JIRA_URL/rest/api/2/issue/$ISSUE_KEY/attachments")

                    if [ "$ATTACH_STATUS" -ne 200 ]; then
                        echo "❌ Attachment upload failed (HTTP $ATTACH_STATUS)"
                        exit 1
                    fi

                    echo "✅ Attachment uploaded successfully"
                '''
            }
        }
    }
}
