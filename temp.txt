import jenkins.model.Jenkins
import com.michelin.cio.hudson.plugins.rolestrategy.*

def strategy = Jenkins.instance.authorizationStrategy

println "Authorization strategy class:"
println strategy.getClass().name
println ""

// HARD fallback: access internal roleMaps directly
def roleMapsField = strategy.getClass().getDeclaredField("roleMaps")
roleMapsField.setAccessible(true)
def roleMaps = roleMapsField.get(strategy)

roleMaps.each { roleType, roleMap ->
    println "=============================="
    println "Role type: ${roleType}"
    println "=============================="

    roleMap.getRoles().each { roleName, role ->
        println "Role: ${roleName}"

        if (role.hasProperty("pattern")) {
            println "  Pattern: ${role.pattern}"
        }

        println "  Permissions:"
        role.permissions.each { p ->
            println "    - ${p.id}"
        }

        println "  Assigned to:"
        roleMap.getRoleEntries()[role].each { sid ->
            println "    - ${sid}"
        }
        println ""
    }
}
