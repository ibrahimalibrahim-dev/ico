import jenkins.model.Jenkins
import com.michelin.cio.hudson.plugins.rolestrategy.*

def strategy = Jenkins.instance.authorizationStrategy

println "Authorization strategy class: ${strategy.getClass().name}"

if (!(strategy instanceof RoleBasedAuthorizationStrategy)) {
    println "Not using Role Strategy Plugin"
    return
}

def roleMaps = [
    GLOBAL : strategy.getRoleMap(RoleType.Global),
    PROJECT: strategy.getRoleMap(RoleType.Project),
    SLAVE  : strategy.getRoleMap(RoleType.Slave)
]

roleMaps.each { type, roleMap ->
    if (roleMap == null) return

    println "\n=============================="
    println "Role type: ${type}"
    println "=============================="

    roleMap.getRoles().each { roleName, role ->
        println "Role: ${roleName}"

        if (role.hasProperty('pattern')) {
            println "  Pattern: ${role.pattern}"
        }

        println "  Permissions:"
        role.permissions.each { p ->
            println "    - ${p.id}"
        }

        println "  Assigned to:"
        roleMap.getRoleEntries()[role].each { sid ->
            println "    - ${sid}"
        }
    }
}
