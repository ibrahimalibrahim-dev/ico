# Create Jira issue and store key
ISSUE_KEY=$(jq -n \
  --arg summary "Automated ticket – Build #${BUILD_NUMBER}" \
  --arg desc "Created by Jenkins pipeline. Build #${BUILD_NUMBER}, Job: ${JOB_NAME}, Target: ${target}" \
  '{
      fields: {
          project: { key: "PROJ" },
          summary: $summary,
          description: $desc,
          issuetype: { name: "Task" }
      }
  }' \
  | curl -s -X POST \
      -H "Authorization: Bearer YOUR_TOKEN" \
      -H "Content-Type: application/json" \
      -d @- \
      https://jira.company.com/rest/api/2/issue \
  | jq -r '.key')

# Exit if ISSUE_KEY is empty
if [ -z "$ISSUE_KEY" ] || [ "$ISSUE_KEY" = "null" ]; then
    echo "❌ Jira issue creation failed"
    exit 1
fi

echo "✅ Created Jira issue: $ISSUE_KEY"

# Upload attachment
curl -X POST \
  -H "Authorization: Bearer YOUR_TOKEN" \
  -H "X-Atlassian-Token: no-check" \
  -F "file=@report.pdf" \
  https://jira.company.com/rest/api/2/issue/$ISSUE_KEY/attachments \
  -w "\nAttachment uploaded for $ISSUE_KEY\n"
