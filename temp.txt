script {

    step([$class: 'DeployApplication',
        siteName: 'UCD',
        deployApp: 'APP',
        deployEnv: 'ENV',
        deployProc: 'PROC',
        // other parameters
    ])

    def logLines = currentBuild.rawBuild.getLog(2000)

    def requestId = null

    logLines.each { line ->
        def matcher = (line =~ /Deployment request id is:\s*([a-zA-Z0-9\-]+)/)

        if (matcher.find()) {
            requestId = matcher.group(1)
        }
    }

    if (requestId) {
        def deploymentUrl = "https://your-ucd-url/#applicationProcessRequest/${requestId}"

        echo "Deployment URL: ${deploymentUrl}"

        currentBuild.description =
            "<a href='${deploymentUrl}'>UCD Deployment</a>"
    }
    else {
        echo "Could not find deployment request id"
    }
}
