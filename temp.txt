#!/bin/bash

# ====== CONFIGURATION ======
BASE_PATH="/opt/apps"              # Change this to your base directory
OWNER="newuser"                    # Change to target user
GROUP="newgroup"                   # Change to target group

# List of folder_app names
FOLDERS=("folder_app1" "folder_app2" "folder_app3")

# ====== SCRIPT LOGIC ======

for folder in "${FOLDERS[@]}"; do
    FULL_PATH="$BASE_PATH/$folder"

    echo "Processing $FULL_PATH..."

    # If folder exists, delete it recursively
    if [ -d "$FULL_PATH" ]; then
        echo "Deleting existing folder: $FULL_PATH"
        rm -rf "$FULL_PATH"
    fi

    # Create folder with nested /app directory
    mkdir -p "$FULL_PATH/app"

    # Change ownership recursively
    chown -R "$OWNER:$GROUP" "$FULL_PATH"

    echo "Created $FULL_PATH/app and set ownership to $OWNER:$GROUP"
    echo "--------------------------------------"
done

echo "All folders processed successfully."
